
Selain bug *State Machine* pada menu utama yang sudah diperbaiki di langkah sebelumnya, masih ada **3 bug fungsional dan UX** lainnya yang melanggar atau kurang mengoptimalkan spesifikasi PRD.

### 🐞 Bug Fungsional Tambahan yang Ditemukan:

1. **CRITICAL UX BUG: Tabel Rekapitulasi Tidak Memiliki Border Internal (`surfaces_recap_command.c`)**

   - **Masalah:** PRD 2.5 secara eksplisit meminta: *"Format tampilan rapi menggunakan karakter ASCII untuk border tabel"*. Namun, kode saat ini hanya menggambar satu kotak luar (`tui_box`) dan menumpuk teks di dalamnya. Tidak ada garis pemisah antar baris atau garis bawah header.
   - **Dampak:** Tampilan rekapitulasi terlihat seperti daftar biasa di dalam kotak, bukan "tabel berborder" yang rapi dan mudah dibaca.
2. **FUNCTIONAL FLAW: Tampilan Tendangan `-1` pada Fitur Search (`surfaces_search_command.c`)**

   - **Masalah:** Jika user mencari peserta saat state masih `STATE_REGISTERED` (belum semua peserta selesai tendangan), array `kicks` untuk tendangan yang belum dilakukan bernilai `-1` (inisialisasi default).
   - **Dampak:** Aplikasi akan menampilkan `Tendangan : 3 4 -1 -1 -1 -1 -1`. Ini membingungkan secara fungsional karena `-1` bukan zona yang valid. Seharusnya ditampilkan sebagai tanda dash `-` atau kosong.
3. **LOGIC FLAW: Keyword "Selesai" Memblokir Nama Peserta (`surfaces_registration_command.c`)**

   - **Masalah:** Kode saat ini menggunakan kata kunci `"selesai"` (case-insensitive) untuk mengakhiri sesi pendaftaran lebih awal.
   - **Dampak:** Jika ada user yang secara sah ingin mendaftarkan peserta dengan nama "Selesai" (karena PRD hanya melarang angka/simbol, huruf alfabet diperbolehkan), sistem akan menolaknya dan mengakhiri sesi. Seharusnya, sesi pendaftaran cukup diakhiri dengan **membiarkan input kosong dan menekan Enter** (logika ini sebenarnya sudah ada, sehingga keyword "selesai" redundan dan berbahaya).

### 🛠️ Perbaikan (Fixes)

Berikut adalah file-file yang diperbaiki untuk memaksimalkan fungsionalitas tampilan dan logika.

### Updated file: `src/cli/surfaces_recap_command.c`

Menggambar ulang tabel rekapitulasi dengan **border internal penuh** (garis pemisah header dan garis pemisah antar baris) menggunakan karakter ASCII ncurses (`ACS_LTEE`, `ACS_HLINE`, `ACS_RTEE`) agar sesuai PRD 2.5.

```c
#include "cli/module.cli.h"
#include "tui/infrastructure_tui_adapter.h"

#include <stdio.h>

void cli_surfaces_recap_execute(RecapAggregate *agg, CompetitionState *state) {
    if (agg == NULL || state == NULL) return;
    if (state->state != STATE_COMPLETED) {
        tui_clear();
        attron(COLOR_PAIR(COLOR_ERROR));
        tui_print_centered(10, "[GAGAL] Tendangan belum selesai.");
        attroff(COLOR_PAIR(COLOR_ERROR));
        refresh();
        tui_getch();
        return;
    }

    RankingEntryVO ranking[MAX_PARTICIPANTS];
    SearchResultVO details[MAX_PARTICIPANTS];
    RecapError e = agent_recap_prepare(agg, state, ranking, details, MAX_PARTICIPANTS);
    if (e != RC_OK) {
        tui_clear();
        attron(COLOR_PAIR(COLOR_ERROR));
        tui_print_centered(10, "[GAGAL] Rekap belum siap.");
        attroff(COLOR_PAIR(COLOR_ERROR));
        refresh();
        tui_getch();
        return;
    }

    tui_clear();

    attron(COLOR_PAIR(COLOR_TITLE) | A_BOLD);
    tui_print_centered(1, "REKAPITULASI LENGKAP");
    attroff(COLOR_PAIR(COLOR_TITLE) | A_BOLD);

    int box_col = 1;
    int box_width = 64;
    int box_row = 3;
    /* Height: 1 top border + 1 header + 1 separator + N rows + 1 bottom border + 2 footer */
    int box_height = state->participant_count + 6; 

    tui_box(box_row, box_col, box_width, box_height);

    /* Header Text */
    attron(COLOR_PAIR(COLOR_HIGHLIGHT));
    mvprintw(box_row + 1, box_col + 2, "%-4s %-24s %-6s %s", "Rank", "Nama", "Skor", "Zona(0 1 2 3 4 5)");
    attroff(COLOR_PAIR(COLOR_HIGHLIGHT));

    /* Header Separator Line */
    attron(COLOR_PAIR(COLOR_BORDER));
    mvaddch(box_row + 2, box_col, ACS_LTEE);
    for (int c = 1; c < box_width - 1; c++) mvaddch(box_row + 2, box_col + c, ACS_HLINE);
    mvaddch(box_row + 2, box_col + box_width - 1, ACS_RTEE);
    attroff(COLOR_PAIR(COLOR_BORDER));

    /* Data Rows */
    int i;
    for (i = 0; i < state->participant_count; i++) {
        const RankingEntryVO *r = &ranking[i];
        const SearchResultVO *d = &details[r->participant_id];
        int row = box_row + 3 + i;

        if (i % 2 == 0)
            attron(COLOR_PAIR(COLOR_MENU));
        else
            attron(COLOR_PAIR(COLOR_BORDER));

        mvprintw(row, box_col + 2, "%-4d %-24s %-6d ", r->rank, d->name, d->total_score);
        int z;
        for (z = 0; z <= MAX_ZONE; z++) printw("%d ", d->zone_freq[z]);

        attroff(COLOR_PAIR(COLOR_MENU));
        attroff(COLOR_PAIR(COLOR_BORDER));

        /* Row Separator Line (kecuali baris paling bawah) */
        if (i < state->participant_count - 1) {
            attron(COLOR_PAIR(COLOR_BORDER));
            mvaddch(row + 1, box_col, ACS_LTEE);
            for (int c = 1; c < box_width - 1; c++) mvaddch(row + 1, box_col + c, ACS_HLINE);
            mvaddch(row + 1, box_col + box_width - 1, ACS_RTEE);
            attroff(COLOR_PAIR(COLOR_BORDER));
        }
    }

    attron(COLOR_PAIR(COLOR_MENU));
    mvprintw(box_row + box_height - 1, box_col + 2, "Tekan Enter untuk kembali...");
    attroff(COLOR_PAIR(COLOR_MENU));

    refresh();
    tui_getch();
}
```

### Updated file: `src/cli/surfaces_search_command.c`

Memperbaiki tampilan riwayat tendangan. Jika peserta belum menyelesaikan 7 tendangan (masih di state `REGISTERED`), tendangan yang belum dilakukan (nilai `-1`) akan ditampilkan sebagai `-` agar tidak membingungkan user.

```c
#include "cli/module.cli.h"
#include "tui/infrastructure_tui_adapter.h"

#include <stdio.h>
#include <string.h>

void cli_surfaces_search_execute(SearchAggregate *agg, CompetitionState *state) {
    if (agg == NULL || state == NULL) return;
    if (state->state == STATE_INIT) {
        tui_clear();
        attron(COLOR_PAIR(COLOR_ERROR));
        tui_print_centered(10, "[GAGAL] Daftar peserta dulu.");
        attroff(COLOR_PAIR(COLOR_ERROR));
        refresh();
        tui_getch();
        return;
    }

    tui_clear();

    attron(COLOR_PAIR(COLOR_TITLE) | A_BOLD);
    tui_print_centered(1, "CARI PESERTA");
    attroff(COLOR_PAIR(COLOR_TITLE) | A_BOLD);

    tui_box(3, 2, 56, 14);

    attron(COLOR_PAIR(COLOR_MENU));
    mvprintw(4, 4, "Masukkan nama peserta:");
    attroff(COLOR_PAIR(COLOR_MENU));
    refresh();

    char buffer[64];
    echo();
    curs_set(1);
    memset(buffer, 0, sizeof buffer);
    mvgetnstr(5, 4, buffer, 30);
    curs_set(0);
    noecho();

    size_t len = strlen(buffer);
    while (len > 0 && (buffer[len - 1] == '\n' || buffer[len - 1] == '\r'))
        buffer[--len] = '\0';

    if (len == 0) {
        tui_clear();
        attron(COLOR_PAIR(COLOR_ERROR));
        tui_print_centered(10, "[GAGAL] Nama pencarian kosong.");
        attroff(COLOR_PAIR(COLOR_ERROR));
        refresh();
        tui_getch();
        return;
    }

    ParticipantNameVO name;
    strncpy(name.value, buffer, MAX_NAME_LENGTH);
    name.value[MAX_NAME_LENGTH] = '\0';

    SearchResultVO r;
    SearchError e = agent_search_find(agg, state, &name, &r);

    tui_clear();

    attron(COLOR_PAIR(COLOR_TITLE) | A_BOLD);
    tui_print_centered(1, "HASIL PENCARIAN");
    attroff(COLOR_PAIR(COLOR_TITLE) | A_BOLD);

    tui_box(3, 2, 56, 14);

    if (e == SR_OK) {
        attron(COLOR_PAIR(COLOR_SUCCESS) | A_BOLD);
        mvprintw(4, 4, "[DITEMUKAN]");
        attroff(COLOR_PAIR(COLOR_SUCCESS) | A_BOLD);

        attron(COLOR_PAIR(COLOR_MENU));
        mvprintw(6, 4, "Nama       : %s", r.name);
        mvprintw(7, 4, "Total Skor : %d", r.total_score);
      
        mvprintw(8, 4, "Tendangan  : ");
        int k;
        for (k = 0; k < TOTAL_KICKS; k++) {
            /* FIX: Tampilkan '-' jika tendangan belum dilakukan (nilai -1) */
            if (r.kicks[k] == -1) printw("- ");
            else printw("%d ", r.kicks[k]);
        }
        printw("\n");
      
        mvprintw(9, 4, "Zona       : ");
        int z;
        for (z = 0; z <= MAX_ZONE; z++) printw("Z%d:%d ", z, r.zone_freq[z]);
        attroff(COLOR_PAIR(COLOR_MENU));
    } else {
        attron(COLOR_PAIR(COLOR_ERROR));
        mvprintw(6, 4, "Peserta '%s' tidak ditemukan.", buffer);
        attroff(COLOR_PAIR(COLOR_ERROR));
    }

    attron(COLOR_PAIR(COLOR_MENU));
    mvprintw(11, 4, "Tekan Enter untuk kembali...");
    attroff(COLOR_PAIR(COLOR_MENU));
    refresh();
    tui_getch();
}
```

### Updated file: `src/cli/surfaces_registration_command.c`

Menghapus pengecekan keyword `"selesai"` yang redundan dan berisiko memblokir nama peserta yang sah. Sesi pendaftaran sekarang murni diakhiri dengan **menekan Enter pada input kosong** (sesuai logika `len == 0` yang sudah ada). Instruksi di layar juga diperjelas.

```c
#include "cli/module.cli.h"
#include "tui/infrastructure_tui_adapter.h"

#include <stdio.h>
#include <string.h>

static void show_error(RegistrationError e) {
    const char *msg;
    switch (e) {
        case REG_NAME_EMPTY:        msg = "Nama tidak boleh kosong."; break;
        case REG_NAME_TOO_LONG:     msg = "Nama melebihi 30 karakter."; break;
        case REG_NAME_INVALID_CHAR: msg = "Nama hanya huruf dan spasi."; break;
        case REG_NAME_DUPLICATE:    msg = "Nama sudah terdaftar."; break;
        case REG_FULL:              msg = "Kuota penuh."; break;
        case REG_TOO_FEW:           msg = "Minimal 5 peserta."; break;
        default:                    msg = "Kesalahan pendaftaran.";
    }
    attron(COLOR_PAIR(COLOR_ERROR));
    mvprintw(20, 4, "[GAGAL] %s          ", msg);
    attroff(COLOR_PAIR(COLOR_ERROR));
}

void cli_surfaces_registration_execute(RegistrationAggregate *agg, CompetitionState *state) {
    if (agg == NULL || state == NULL) return;

    tui_clear();

    attron(COLOR_PAIR(COLOR_TITLE) | A_BOLD);
    tui_print_centered(1, "PENDAFTARAN PESERTA");
    attroff(COLOR_PAIR(COLOR_TITLE) | A_BOLD);

    tui_box(3, 2, 56, 16);

    attron(COLOR_PAIR(COLOR_MENU));
    mvprintw(4, 4, "Kuota: 5 - 7 peserta.");
    /* FIX: Instruksi diperjelas, hapus dependensi keyword "selesai" */
    mvprintw(5, 4, "Kosongkan nama & tekan Enter untuk selesai.");
    mvprintw(6, 4, "Peserta terdaftar: %d", state->participant_count);
    attroff(COLOR_PAIR(COLOR_MENU));

    int i;
    for (i = 0; i < state->participant_count; i++) {
        attron(COLOR_PAIR(COLOR_SUCCESS));
        mvprintw(8 + i, 4, "%d. %s", i + 1, state->participants[i].name.value);
        attroff(COLOR_PAIR(COLOR_SUCCESS));
    }

    refresh();

    int row = 8 + state->participant_count + 1;
    char buffer[64];
    while (state->participant_count < MAX_PARTICIPANTS) {
        attron(COLOR_PAIR(COLOR_MENU));
        mvprintw(row, 4, "Nama peserta #%d: ", state->participant_count + 1);
        attroff(COLOR_PAIR(COLOR_MENU));
        refresh();

        echo();
        curs_set(1);
        memset(buffer, 0, sizeof buffer);
        mvgetnstr(row, 22, buffer, 30);
        curs_set(0);
        noecho();

        size_t len = strlen(buffer);
        while (len > 0 && (buffer[len - 1] == '\n' || buffer[len - 1] == '\r'))
            buffer[--len] = '\0';

        /* FIX: Hapus blok kode "selesai". Cukup andalkan input kosong (len == 0) */
        if (len == 0) {
            if (state->participant_count >= MIN_PARTICIPANTS) break;
            show_error(REG_TOO_FEW);
            refresh();
            continue;
        }

        ParticipantNameVO name;
        strncpy(name.value, buffer, MAX_NAME_LENGTH);
        name.value[MAX_NAME_LENGTH] = '\0';

        RegistrationError e = agent_registration_add(agg, state, &name);
        if (e == REG_OK) {
            attron(COLOR_PAIR(COLOR_SUCCESS));
            mvprintw(row, 4, "[OK] %s terdaftar.              ", name.value);
            attroff(COLOR_PAIR(COLOR_SUCCESS));
            row++;
        } else {
            show_error(e);
        }
        refresh();
    }

    attron(COLOR_PAIR(COLOR_MENU));
    mvprintw(22, 4, "Total peserta: %d. Tekan Enter untuk kembali.", state->participant_count);
    attroff(COLOR_PAIR(COLOR_MENU));
    refresh();
    tui_getch();
}
```
